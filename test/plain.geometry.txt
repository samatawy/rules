# Common rules for plain geometry

CONST CIRCLE = 'circle'

CONST RECTANGLE = 'rectangle'

CONST SQUARE = 'square'

CONST TRIANGLE = 'triangle'

circle_area(radius: number) = pi() * radius * radius

square_area(side: number) = side * side

rectangle_area(length: number, height: number) = length * breadth

triangle_area(base: number, height: number) = base * height * 0.5

IF lowerCase(shape.type) == TRIANGLE THEN shape.area = triangle_area(shape.base, shape.height)
